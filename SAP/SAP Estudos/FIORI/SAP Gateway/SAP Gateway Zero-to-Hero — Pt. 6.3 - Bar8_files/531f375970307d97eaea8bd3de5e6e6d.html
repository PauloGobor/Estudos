FUNCTION Z_BAR8_SERVICE_01.
*"--------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" VALUE(CARRID) TYPE S_CARR_ID
*" EXPORTING
*" VALUE(MOST_EXPENSIVE) TYPE S_PRICE
*"--------------------------------------------------------------------
SELECT SINGLE price
FROM sflight
INTO most_expensive
WHERE carrid = CARRID.
ENDFUNCTION.
